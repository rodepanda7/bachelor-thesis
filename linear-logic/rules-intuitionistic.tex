		\end{prooftree}

		%CONTRACTION
		\begin{prooftree}
			\def\extraVskip{5pt}
			\AxiomC{$\judgement{\Gamma, y : A, z : A}{u : B}$}
			\RightLabel{Contraction}
			\UnaryInfC{$\judgement{\Gamma, x : A}{u \subst{y}{x}\subst{z}{x} : B}$} % Conclusion
		\end{prooftree}
		%WEAKENING

			\def\extraVskip{5pt}

		\end{prooftree}

			\def\extraVskip{5pt}
